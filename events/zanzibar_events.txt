namespace = zanzibar

# For OMA
zanzibar.1 = {
	type = country_event
	placement = scope:sultanate_of_zanzibar_capital

	title = zanzibar.1.t
	desc = zanzibar.1.d
	flavor = zanzibar.1.f

	gui_window = event_window_1char_tabloid

	left_icon = scope:sultan_of_zanzibar

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	trigger = {
		# triggered by journal entry
	}

	immediate = {
		save_scope_as = muscat_and_oman
		ruler = {
			save_scope_as = sultan_of_muscat_and_oman
		}
		if = {
			limit = {
				any_scope_state = {
					state_region = s:STATE_OMAN
				}
			}
			set_capital = STATE_OMAN
		}
		create_country = {
			tag = ZAN
			origin = this
			state = s:STATE_ZANZIBAR.region_state:OMA
			on_created = {
				create_character = {
					first_name = "Majid"					
					last_name = "al_Said"
					historical = yes
					culture=cu:bedouin
					ruler = yes
					traits = { hedonist ambitious }
					birth_date = 1834.10.19 # Majid's exact birthdate is unknown, so it is arbitrarily being set to the same month and day as the start of his reign as sultan, because a January 1st birthday is boring.
				}
			}
		}
		c:ZAN = { 
			save_scope_as = sultanate_of_zanzibar
			ruler = {
				save_scope_as = sultan_of_zanzibar
			}
	 		capital = {
	 			save_scope_as = sultanate_of_zanzibar_capital
	 		}
		}
	}

	option = {
		name = zanzibar.1.a
		custom_tooltip = zanzibar.1.a.tt
		default_option = yes
		
		# Cede East African holdings to Zanzibar
		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_SOMALIA #Oman keeps STATE_SOMALIALAND (control of the Red Sea)
					state_region = s:STATE_OGADEN #Oman keeps Ethiopia
					region = sr:region_african_lakes #Great Lakes
					region = sr:region_zanj
					region = sr:region_zambezia
					region = sr:region_madagascar
					region = sr:region_southern_africa
					region = sr:region_south_west_africa
					region = sr:region_congo
				}
			}
			set_state_owner = c:ZAN
		}
		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_SOMALIA #Oman keeps STATE_SOMALIALAND (control of the Red Sea)
					state_region = s:STATE_OGADEN #Oman keeps Ethiopia
					region = sr:region_african_lakes #Great Lakes
					region = sr:region_zanj
					region = sr:region_zambezia
					region = sr:region_madagascar
					region = sr:region_southern_africa
					region = sr:region_south_west_africa
					region = sr:region_congo
				}
				state_region = {
					has_claim_by = c:OMA
				}
			}
			add_claim = c:ZAN
		}
	}
	
	option = {
		name = zanzibar.1.b
		
		trigger = {
			is_player = yes # no point in this option for the AI
		}
		
		play_as = c:ZAN
		
		# Cede East African holdings to Zanzibar
		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_SOMALIA #Oman keeps STATE_SOMALIALAND (control of the Red Sea)
					state_region = s:STATE_OGADEN #Oman keeps Ethiopia
					region = sr:region_african_lakes #Great Lakes
					region = sr:region_zanj
					region = sr:region_zambezia
					region = sr:region_madagascar
					region = sr:region_southern_africa
					region = sr:region_south_west_africa
					region = sr:region_congo
				}
			}
			set_state_owner = c:ZAN
		}
		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_SOMALIA #Oman keeps STATE_SOMALIALAND (control of the Red Sea)
					state_region = s:STATE_OGADEN #Oman keeps Ethiopia
					region = sr:region_african_lakes #Great Lakes
					region = sr:region_zanj
					region = sr:region_zambezia
					region = sr:region_madagascar
					region = sr:region_southern_africa
					region = sr:region_south_west_africa
					region = sr:region_congo
					state_region = {
						has_claim_by = c:OMA
					}
				}
			}
			add_claim = c:ZAN
		}
	}
	
	option = {
		name = zanzibar.1.c

		custom_tooltip = zanzibar.1.a.tt
		
		# Don't cede East African holdings to Zanzibar, and start a fight
		
		s:STATE_ZANZIBAR = {
			add_claim = c:OMA
		}

		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_SOMALIA #Oman keeps STATE_SOMALIALAND (control of the Red Sea)
					state_region = s:STATE_OGADEN #Oman keeps Ethiopia
					region = sr:region_african_lakes #Great Lakes
					region = sr:region_zanj
					region = sr:region_zambezia
					region = sr:region_madagascar
					region = sr:region_southern_africa
					region = sr:region_south_west_africa
					region = sr:region_congo
				}
			}
			state_region = {
				add_claim = c:ZAN
			}
		}
	}
}
