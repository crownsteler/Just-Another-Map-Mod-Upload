cr_indian_removal.1 = { #The Chickasaw Removal
	type = country_event
	placement = p:x3B0B58

	title = cr_indian_removal.1.t
	desc = cr_indian_removal.1.desc
	flavor = cr_indian_removal.1.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	trigger = {
		NOT = { has_variable = cr_treaty_of_doaksville }
		c:CSW = {
			any_scope_state = {
				state_region = s:STATE_MISSISSIPPI
			}
		}
		OR = {
			is_ai = yes
			c:CSW = {
				relations:ROOT <= -20
			}
		}
	}
	
	immediate = {
		set_variable = cr_treaty_of_doaksville
		s:STATE_OKLAHOMA = {
			set_owner_of_provinces = {
				country = c:CSW
				provinces = {
					x32D681 xBFD5C4 x4BE9C6
				}
			}
		}	
		c:CSW = { set_capital = STATE_OKLAHOMA }
		s:STATE_OKLAHOMA.region_state:CSW = {
			every_scope_pop = {
				limit = {
					OR = {
						culture = cu:comanche							
					}
				}
				move_pop = s:STATE_OKLAHOMA.region_state:COM 
			}
		}
	}
	
	option = {
		name = cr_indian_removal.1.a
		#custom_tooltip = cr_indian_removal.1.a_tt
		add_modifier = {
			name = cr_treaty_of_doaksville
			months = 49
		}
		if = {
			limit = {
				any_political_movement = {
					is_political_movement_type = movement_to_enact
					desired_law = law_type:law_no_colonial_affairs
					NOT = { has_modifier = cr_laws_indian_removal_paid_chickasaw_law }
				}
			}
			random_political_movement = {
				limit = {
					is_political_movement_type = movement_to_enact
					desired_law = law_type:law_no_colonial_affairs
				}
				add_modifier = {
					name = cr_laws_indian_removal_paid_chickasaw_law
				}
			}
		}
		if = {
			limit = {
				NOT = {
					has_journal_entry = je_chickasaw_removal
				}
			}
			add_journal_entry = { type = je_chickasaw_removal }
		}
		hidden_effect = {
			set_variable = cr_chickasaw_removal_has_started
		}
		set_variable = cr_treaty_of_doaksville_debt_to_be_paid
		ai_chance = {
			base = 20
			modifier = {
				scaled_debt > 0.3
				add = -20
			}
		}
	}
	
	option = {
		name = cr_indian_removal.1.b
		default_option = yes
		custom_tooltip = cr_indian_removal.1.b_tt
		change_relations = {
			country = c:CSW
			value = -20
		}
		if = {
			limit = {
				any_political_movement = {
					is_political_movement_type = movement_to_enact
					desired_law = law_type:law_no_colonial_affairs
					NOT = { has_modifier = cr_laws_indian_removal_not_paid_chickasaw_movement }
				}
			}
			random_political_movement = {
				limit = {
					is_political_movement_type = movement_to_enact
					desired_law = law_type:law_no_colonial_affairs
				}
				add_modifier = {
					name = cr_laws_indian_removal_not_paid_chickasaw_movement
				}
			}
		}
		else = {
			add_radicals = {
				value = small_radicals
				culture = cu:yankee
				interest_group = ig:ig_intelligentsia
			}
		}
		if = {
			limit = {
				NOT = {
					has_journal_entry = je_chickasaw_removal
				}
			}
			add_journal_entry = { type = je_chickasaw_removal }
		}
		hidden_effect = {
			set_variable = cr_chickasaw_removal_has_started
		}
		ai_chance = {
			base = 80
		}
	}
}
