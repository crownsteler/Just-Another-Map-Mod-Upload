cr_indian_removal.3 = { #The Chickasaw Removal
	type = country_event
	placement = scope:capital_scope

	title = cr_indian_removal.3.t
	desc = {
		first_valid = {
			triggered_desc = {
				desc = cr_indian_removal.3.desc_1
				trigger = {
					exists = c:CSW
				}
			}
			triggered_desc = {
				desc = cr_indian_removal.3.desc_2
				trigger = {
					always = yes
				}
			}
		}
	}
	flavor = cr_indian_removal.3.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	#on_opened_soundeffect = "event:/SFX/Events/unspecific/trains"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	trigger = {
		year >1865
		has_variable = cr_treaty_of_doaksville_debt_to_be_paid
		is_at_war = no
		#something with ruling party or ruler ideology
	}
	
	immediate = {
		c:CSW = {
			save_scope_as = chickasaw_nation
		}
		if = {
			limit = {
				exists = c:CSW
			}
			c:CSW.capital = {
				save_scope_as = capital_scope
			}
		}
		else_if = {
			ROOT.capital = {
				save_scope_as = capital_scope
			}
		}
		s:STATE_MISSISSIPPI.region_state:USA = {
			save_scope_as = state_of_mississippi
		}
		
	}
	
	option = {
		name = cr_indian_removal.3.a
		remove_variable = cr_treaty_of_doaksville_debt_to_be_paid
		add_modifier = {
			name = cr_treaty_of_doaksville
			months = 73
		}
		if = {
			limit = {
				exists = c:CSW
			}
			change_relations = {
				country = c:CSW
				value = 15
			}
			create_diplomatic_pact = {
				country = c:CSW
				type = puppet
			}
		}
		ai_chance = {
			base = 50
			modifier = {
				if = {
					scaled_debt > 0.5
				}
				add = -50
			}
		}
	}
	
	option = {
		name = cr_indian_removal.3.b
		default_option = yes
		remove_variable = cr_treaty_of_doaksville_debt_to_be_paid
		add_radicals = {
			value = small_radicals
			culture = cu:yankee
			interest_group = ig:ig_intelligentsia
		}
		add_radicals = {
			value = very_large_radicals
			culture = cu:caddoan
		}
		if = {
			limit = {
				exists = c:CSW
			}
			change_relations = {
				country = c:CSW
				value = -50
			}
			create_diplomatic_pact = {
				country = c:CSW
				type = puppet
			}
		}
		ai_chance = {
			base = 50
		}
	}
}
